400  template <typename T> size_t Variant::get(T &v) const



401  {



402    return Type::ID<typename T::value_type>() == type_id() ?



403      (v = T((typename T::const_pointer)begin(), (typename T::const_pointer)end())).size() : 0;



404  }



405



406  //// Event ////



407



408  template <typename T> size_t Event::size() const



409  { return Type::ID<T>() == type_id() ? ((const char*)_data_end - (const char*)_data) / sizeof(T) : 0; }



410



411  template <typename T> size_t Event::find(const std::string &name, T &v) const



412  {



413    if(type_id() == Type::FRAME)



414      for(const Event *e = begin(); e != end(); e++)



415        if(Type::ID<typename T::value_type>() == e->type_id() && name == e->name())



416          return (v = T((typename T::const_pointer)e->begin(), (typename T::const_pointer)e->end())).size();



417    return 0;



418  }

422#define OWL_TYPE_OPERATORS(T1, T2)                                      \



423  template <> inline Type::operator T1() const                          \



424  { return ID<T1>() == id && (const T1*)data ? *(const T1*)data : ID<T2>() == id && (const T2*)data ? *(const T2*)data : T1(); }



425



426#define OWL_VARIANT_VECTOR_OPERATORS(V, T1, T2)                         \



427  template <> inline Variant::operator V<T1>() const                    \



428  { return Type::ID<T2>() == type_id() ? V<T1>((const T2*)begin(), (const T2*)end()) : V<T1>((const T1*)begin(), (const T1*)end()); }
