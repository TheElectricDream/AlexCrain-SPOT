44    struct Frame : std::vector<Event> {



45      uint16_t id;



46      int64_t time;

66    SocketDevice() : port(0)

71      udp.verbose = 1;



72      udp.out = 0;

86    int open(const std::string &device, const std::string &device_options)



87    {



88      if(sock && sock.getf(CONNECTED)) return 1;



89



90      strings a = split(device, ':');



91      stringmap o(device_options);



92



93      const char *addr = a.size() > 0 && !a[0].empty() ? a[0].c_str() : "localhost";



94      port = a.size() > 1 ? strtoi(a[1]) : 0;



95      int t = get(o, "timeout", int(3000000));



96



97      if(!sock)



98        {



99          int ret = sock.connect(addr, 8000+port, t);



100          update_error();



101          if(ret <= 0) return ret;



102        }



103      else if(!sock.getf(CONNECTED))



104        {



105          int ret = sock.connected(t);



106          update_error();



107          if(ret <= 0) return ret;



108        }



109



110      { // use tcp socket's port# as udp port#



111        int port = 0;



112        if(const char *address = sock.getsaddr(&port)) udp.bind(address, port);



113      }



114



115      sock.recv_bufsize(16*1024*1024);



116      sock.send_bufsize(16*1024*1024);



117



118      sock.nonblock(false);



119      sock.setf(CONNECTED);



120      update_error();



121



122      time = -1;



123



124      return 1;



125    }



126



127    bool close()



128    {



129      if(!sock) return false;



130      events.clear();



131



132      sock.close();



133      udp.close();



134      broadcast.close();



135



136      sock.clear();



137      udp.clear();



138      broadcast.clear();



139



140      update_error();



141



142      return true;



143    }



144



145    bool is_open() const



146    { return sock ? sock.getf(CONNECTED) : false; }



147



148    int read(long timeout)



149    {



150      select.clear() << sock.sock() << udp.sock() << broadcast.sock();



151      timeval tv = {timeout / 1000000, timeout % 1000000};



152      int ret = select(tv, 1);



153      if(ret <= 0) return ret;



154



155      ret = 0;



156      while(sock && sock.select(0, 1) > 0 && recv(sock) > 0) ret++;



157      while(udp && udp.select(0, 1) > 0 && recv(udp) > 0) ret++;



158      while(broadcast && broadcast.select(0, 1) > 0 && recv(broadcast) > 0) ret++;



159



160      return ret ? ret : (sock ? 0 : -1);



161    }



162



163    int recv(basic_socket &s)



164    {



165      Header h;



166      packet p;



167



168      int ret = OWL::recv(s, h, p);



169      update_error();



170      if(ret <= 0) return ret;



171



172      int count = 0;



173      // multi-message packet



174      while(p.index <= p.size())



175        {



176          ret = recv(h, p);



177          if(ret <= 0) return ret;



178          count++;



179          p.index += h.size;



180



181          if(p.size() < p.index + sizeof(Header)) break;



182          h = *(Header*)(p.data()+p.index);



183          p.index += sizeof(Header);



184        }



185      return count;



186    }



187



188    int recv(Header &h, packet &p)



189    {



190      if(time < h.time) time = h.time;



191



192      if(h.type == Type::BYTE && (h.id == INITIALIZE || h.id == DONE))



193        {



194          stringmap m(string(p.data()+p.index, h.size));



195          vector<int> n;



196          if(get(m, "streaming", n)) toggle_broadcast(n[0]);



197        }



198      else if(h.type == Type::INT && h.id == STREAMING && h.size == 4)



199        {



200          const int *v = (const int*)(p.data()+p.index);



201          toggle_broadcast(*v);



202        }



203



204      // frame events have frame id and event ids for transmission: (frame.id << 8) | event.id



205      uint16_t frame_id = (h.id >> 8);



206      if(frame_id)



207        {



208          // accumulate frame events



209          if(frame_id != newFrame.id || h.time != newFrame.time) newFrame = Frame(frame_id, h.time);



210          newFrame.push_back(decode(p, h.size, h.type, (h.id & 0xFF), 0, h.time));



211          if(!newFrame.back().valid()) { newFrame.pop_back(); return 0; }



212        }



213      else if(h.type == Type::FRAME && (p.size() <= p.index) && h.id == newFrame.id && h.time == newFrame.time)



214        {



215          // copy frame events into frame



216          events.push_back(EventPrivate(h.type, h.id, 0, h.time, newFrame));



217          newFrame = Frame();



218        }



219      else



220        {



221          events.push_back(decode(p, h.size, h.type, h.id, 0, h.time));



222          if(!events.back().valid()) { events.pop_back(); return 0; }



223        }



224



225      return 1;



226    }



227



228    int write(uint16_t type, uint16_t id, uint32_t flags, const char *data, size_t size)



229    {



230      if(!sock || !valid(id)) return -1;



231



232      Header h(type, id, size);



233



234      buffer.clear();



235      buffer.insert(buffer.end(), (const char*)&h, (const char*)(&h+1));



236      if(size) buffer.insert(buffer.end(), data, data+size);



237



238      int ret = sock.send(buffer.data(), buffer.size());



239      update_error();



240      if(ret != (int)buffer.size()) return -1;



241



242      return ret;



243    }



244



245    void update_error()



246    {



247      error = err.str();



248      err.str("");



249      if(!error.empty() && error[error.size()-1] == '\n') error.erase(error.size()-1);



250    }



251



252    void toggle_broadcast(int v)



253    {



254      if(v == 3 && !broadcast)



255        {



256          broadcast.listen(8500+port);



257          broadcast.nonblock();



258          broadcast.broadcast(1);



259          broadcast.recv_bufsize(16*1024*1024);



260          broadcast.send_bufsize(16*1024*1024);



261        }



262      else if(v != 3 && broadcast) broadcast.close();



263      update_error();



264    }



265

267

288  bool Scan::send(const std::string &message)



289  {



290    udp_socket sock(fd);



291    sock.verbose = 0;



292    if(!sock)



293      {



294        sock.socket();



295        if(!sock) return false;



296        sock.nonblock();



297        sock.broadcast(1);



298        fd = sock.sock();



299      }



300    sock.setoaddr(0, 8998);



301



302    // format message?



303    ostring out;



304    char name[1024] = "";



305    gethostname(name, sizeof(name));



306    out << "hostname=" << name;



307    out << " protocol=" << int(OWL_PROTOCOL_VERSION);



308#ifdef LIBOWL_REV



309    out << " libowl=5.1." << LIBOWL_REV;



310#endif // LIBOWL_REV



311    if(!message.empty()) out << " " << message;



312    string s = out;



313    bool ret = sock.send(s.c_str(), s.size()+1) > 0;



314    if(!sock) fd = -1;



315    return ret;



316  }

318  std::vector<std::string> Scan::listen(long timeout)



319  {



320    std::vector<std::string> out;



321    udp_socket sock(fd);



322    sock.verbose = 0;



323    if(!sock) return out;



324



325    int ret = sock.select(timeout, 1);



326    if(!sock) fd = -1;



327    if(ret <= 0) return out;



328



329    vector<Event> events;



330    char buf[9*1024];



331    while(1)



332      {



333        ret = sock.recv(buf, sizeof(buf));



334        if(!sock) fd = -1;



335        if(ret <= 0) break;



336        buf[ret] = 0;



337



338        const char *iaddr = sock.getiaddr();



339        if(!iaddr) continue;



340



341        out.push_back(string("ip=") + iaddr + " " + buf);



342      }



343



344    return out;



345  }
