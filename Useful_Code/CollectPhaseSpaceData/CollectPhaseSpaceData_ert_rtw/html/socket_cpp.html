97inline bool set_flags(size_t &flag, size_t bit, int value=1)



98{



99  // toggle



100  if(value == -1) value = 1 - ((flag>>bit) & 1);



101  flag &= ~(1 << bit);



102  flag |= (value << bit);



103  return value;



104}



105



106inline bool get_flags(size_t flag, size_t bit)



107{



108  return (flag & (1 << bit));



109}

111int get_net_addr(unsigned int *net_addr, const char *netaddress, int family, int type, int protocol)



112{



113  if(!net_addr) return -1;



114  addrinfo hints;



115  addrinfo *res = NULL;



116  memset(&hints, 0, sizeof(hints));



117  hints.ai_flags = AI_ADDRCONFIG;



118  hints.ai_family = family;



119  hints.ai_socktype = type;



120  hints.ai_protocol = protocol;



121  if(int ret = getaddrinfo(netaddress, NULL, &hints, &res)) // reentrant



122    {



123      cerr << "getaddrinfo: " << gai_strerror(ret) << endl;



124      return -1;



125    }



126  *net_addr = ((sockaddr_in*)res->ai_addr)->sin_addr.s_addr;



127  freeaddrinfo(res);



128  return 1;



129}

131// set addr to net_addr and port



132void set_addr(sockaddr *addr, int domain, unsigned int net_addr, int port)



133{



134  if(domain == AF_INET)



135    {



136      sockaddr_in *addr_in = (sockaddr_in*)addr;



137      memset((char*)addr_in, 0, ADDRLEN);



138      addr_in->sin_family = AF_INET;



139      addr_in->sin_port = port;



140      addr_in->sin_addr.s_addr = net_addr;



141    }



142  else



143    {



144      cerr << "set_addr: unsupported domain" << endl;



145      exit(-1);



146    }



147}

149ostream &operator<<(ostream &out, const sockaddr &addr)



150{



151  if(addr.sa_family == AF_INET)



152    {



153      const sockaddr_in *in = (const sockaddr_in*)&addr;



154      out << inet_ntoa(in->sin_addr) << ":" << ntohs(in->sin_port);



155    }



156  return out;



157}

191void basic_socket::nonblock(bool value) const



192{



193#ifdef WIN32



194  ULONG argp = value;



195  if(ioctlsocket(_sock, FIONBIO, &argp))



196    {



197      perror("ioctlsocket");



198      exit(-1);



199    }



200#else // !WIN32



201  int flags = fcntl(_sock, F_GETFL, 0);



202  if(value) flags |= O_NONBLOCK;



203  else flags &= (~O_NONBLOCK);



204  if(fcntl(_sock, F_SETFL, flags) < 0)



205    {



206      perror("fcntl");



207      exit(-1);



208    }



209#endif // WIN32



210}

240int basic_socket::getnetaddr(const char *netaddress, unsigned int *net_addr) const



241{



242  if(!netaddress || !net_addr) return -1;



243  addrinfo hints;



244  addrinfo *res = NULL;



245  memset(&hints, 0, sizeof(hints));



246  hints.ai_flags = AI_ADDRCONFIG;



247  hints.ai_family = domain();



248  hints.ai_socktype = type();



249  hints.ai_protocol = protocol();



250  if(int ret = getaddrinfo(netaddress, NULL, &hints, &res)) // reentrant



251    {



252      derr << "error: " << netaddress << ": " << gai_strerror(ret) << endl;



253      return -1;



254    }



255  *net_addr = ((sockaddr_in*)res->ai_addr)->sin_addr.s_addr;



256  freeaddrinfo(res);



257  return 1;



258}

281// set addr to netaddress:port, or broadcast:port



282int basic_socket::setaddr(sockaddr *addr, const char *netaddress, int port) const



283{



284  unsigned int net_addr = INADDR_ANY;



285



286  if(netaddress)



287    {



288      if(getnetaddr(netaddress, &net_addr) == -1)



289        return -1;



290    }



291  else if(getbroadcast(&net_addr) == -1)



292    {



293      derr << "error: could not get broadcast address" << endl;



294      return -1;



295    }



296



297  set_addr(addr, domain(), net_addr, htons(port));



298



299  return 1;



300}

334// create socket, set options



335int basic_socket::socket()



336{



337  int reuse_addr = 1;



338  int nodelay = 1;



339



340  // create socket



341  _sock = ::socket(domain(), type(), 0);



342  if(_sock < 0)



343    {



344      perror("socket");



345      exit(-1);



346    }



347



348  nonexec(_sock);



349



350  // reuse the address



351  if(reuse_addr && setsockopt(SOL_SOCKET, SO_REUSEADDR, reuse_addr) == -1)



352    {



353      perror("setsockopt(SO_REUSEADDR)");



354      exit(-1);



355    }



356



357  // no delay



358  if(type() == SOCK_STREAM && nodelay && setsockopt(SOL_TCP, TCP_NODELAY, nodelay) == -1)



359    {



360      perror("setsockopt(TCP_NODELAY)");



361      exit(-1);



362    }



363



364  dout << "new socket (" << _sock << ")" << endl;



365



366  return _sock;



367}

379int basic_socket::bind(const char *netaddress, unsigned short port)



380{



381  if(socket() < 0) return _sock;



382



383  int ret = setaddr(&saddr, netaddress, port);



384  if(ret < 0) { close(); return ret; }



385



386  // bind socket to netaddress:port



387  ret = ::bind(_sock, &saddr, ADDRLEN);



388  if(ret < 0)



389    {



390      perror("bind");



391      close();



392      return -1;



393    }



394



395  dout << "bind (" << _sock << "): " << saddr << endl;



396



397  return _sock;



398}

400int basic_socket::listen(unsigned short port)



401{



402  if(socket() < 0) return _sock;



403



404  set_addr(&saddr, domain(), INADDR_ANY, htons(port));

442// nonblocking connect, use with connected()



443// return 0 on waiting



444int basic_socket::connect(const char *netaddress, unsigned short port, long timeout)



445{



446  if(socket() < 0) return _sock;



447



448  if(setoaddr(netaddress, port) < 0) return _sock;



449



450  nonblock();



451



452  int ret = ::connect(_sock, &oaddr, ADDRLEN);



453  if(ret < 0 && IS_NET_ERROR(EINPROGRESS, WSAEWOULDBLOCK) && (ret = connected(timeout)) == 0)



454    return 0; // return 0 on waiting



455  if(ret < 0)



456    {



457      derr << "error: connect to " << netaddress << ": " << NET_ERROR_STR() << endl;



458      close();



459      return ret;



460    }



461



462  return _sock;



463}

465// return



466//  sock on success



467//  0 on waiting



468// -1 on error



469int basic_socket::connected(long timeout)



470{



471  // wait for timeout



472  int ret = select(timeout, 2);



473  if(ret == 0)



474    {



475      // no timeout -- no error



476      if(timeout == 0) return 0;



477



478      // socket not ready -- not connected in time



479      SET_NET_ERROR(ETIMEDOUT, WSAETIMEDOUT);



480      return -1;



481    }



482



483  // have socket, check for any error



484  ret = geterror();



485  if(ret)



486    {



487      if(ret > 0) SET_NET_ERROR(ret, ret);



488      return -1;



489    }



490



491  getsaddr();



492  dout << "connect (" << _sock << "): " << saddr << " to " << oaddr << endl;



493



494  return _sock;



495}

497// flag: 1 - read



498// flag: 2 - write



499// flag: 3 - read+write



500int basic_socket::select(timeval &tv, int flag)



501{



502  if(_sock < 0) return -1;



503  fd_set rfds, wfds;



504  FD_ZERO(&rfds);



505  FD_ZERO(&wfds);



506#ifndef WIN32



507  if(flag & 1) FD_SET(_sock, &rfds);



508  if(flag & 2) FD_SET(_sock, &wfds);



509#else



510  if(flag & 1) FD_SET(u_int(_sock), &rfds);



511  if(flag & 2) FD_SET(u_int(_sock), &wfds);



512#endif



513  int ret = ::select(flag ? _sock+1 : 0, &rfds, &wfds, 0, &tv);



514



515  if(ret < 0)



516    {



517      if(errno == EINTR) return 0;



518      perror("select");



519      close();



520      return ret;



521    }



522



523  return ret;



524}

526int basic_socket::select(long timeout, int flag)



527{



528  if(_sock < 0) return -1;



529  timeval tv = {timeout / 1000000, timeout % 1000000};



530  return select(tv, flag);



531}

546basic_socket basic_socket::accept(long timeout)



547{



548  int new_sock = -1;

576//  0: no data

578int basic_socket::send(const char *buf, size_t count)



579{



580  if(_sock < 0) return -1;



581  if(buf == 0 || count == 0) return 0;



582



583  //int ret = ::send(_sock, buf, count, 0);



584  int ret = sendto(_sock, buf, count, send_flags, &oaddr, ADDRLEN);



585



586  if(ret < 0)



587    {



588      if(IS_NET_ERROR(EAGAIN, -1) || IS_NET_ERROR(EWOULDBLOCK, WSAEWOULDBLOCK)) return 0;



589      derr << "error: sendto (" << _sock << "): " << NET_ERROR_STR() << endl;



590      close();



591      return ret;



592    }



593



594  return ret;



595}

597// return:



598// -1: error



599//  0: no data



600// >0: data received



601int basic_socket::recv(char *buf, size_t count)



602{



603  if(_sock < 0) return -1;



604  if(buf == 0 || count == 0) return 0;



605



606  //int ret = ::recv(_sock, buf, count, 0);



607  socklen_t addrlen = ADDRLEN;



608  int ret = recvfrom(_sock, buf, count, recv_flags, &iaddr, &addrlen);



609



610  if(ret == 0)



611    {



612      derr << "Connection closed by foreign host (" << _sock << ")" << endl;



613      close();



614      return -1;



615    }



616



617  if(ret < 0)



618    {



619      if(IS_NET_ERROR(EAGAIN, -1) || IS_NET_ERROR(EWOULDBLOCK, WSAEWOULDBLOCK)) return 0;



620      derr << "error: recv (" << _sock << "): " << NET_ERROR_STR() << endl;



621      close();



622      return ret;



623    }



624



625  return ret;



626}
